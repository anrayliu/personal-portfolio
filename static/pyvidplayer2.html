# This is a simple demo Python file
import time
import random
class Timer:
def __init__(self):
self.start_time = time.time()
def elapsed(self):
return round(time.time() - self.start_time, 2)
def simulate():
t = Timer()
for i in range(5):
delay = random.uniform(0.5, 1.5)
time.sleep(delay)
print(f"Step {i+1} took {t.elapsed()}s")
if __name__ == "__main__":
simulate()
